<script>
// Animasi cinta
const canvas = document.querySelector('canvas');
const ctx = canvas.getContext('2d');
canvas.width = innerWidth;
canvas.height = innerHeight;
const hearts = [];
class Heart {
constructor() {
this.x = Math.random() * canvas.width;
this.y = canvas.height + Math.random() * 100;
this.size = Math.random() * 20 + 10;
this.speed = Math.random() * 1 + 0.5;
this.alpha = 1;
}
draw() {
ctx.globalAlpha = this.alpha;
ctx.fillStyle = "#ff69b4";
ctx.beginPath();
ctx.moveTo(this.x, this.y);
ctx.bezierCurveTo(this.x - this.size/2, this.y - this.size/2,
this.x - this.size, this.y + this.size/3, this.x, this.y + this.size);
ctx.bezierCurveTo(this.x + this.size, this.y + this.size/3,
this.x + this.size/2, this.y - this.size/2, this.x, this.y);
ctx.fill();
ctx.globalAlpha = 1;
}
update() {
this.y -= this.speed;
this.alpha -= 0.005;
}
}
function animate() {
ctx.clearRect(0, 0, canvas.width, canvas.height);
hearts.push(new Heart());
for (let i = 0; i < hearts.length; i++) {
hearts[i].update();
hearts[i].draw();
if (hearts[i].alpha <= 0) {
hearts.splice(i, 1);
i--;
}
}
requestAnimationFrame(animate);
}
animate();
</script>
